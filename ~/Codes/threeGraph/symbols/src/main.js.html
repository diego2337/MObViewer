 20 /**
 21  * Add number of edges to .json string being created.
 22  * @public
 23  * @param {string} data - .json string.
 24  * @returns {string} .json string containing number of edges.
 25  */
 26 function addNumberOfEdgesToJSON(data)
 27 {
 28   var jason = JSON.parse(data);
 29   jason.graphInfo[0].edges = jason.links.length.toString();
 30   data = JSON.stringify(jason);
 31   return data;
 32 }

 37  * @returns {string} either '\' or '/' symbol for folder paths.
 38  */
 39 function addFolderPath()
 40 {
 41   return process.platform == "win32" ? "\\" : "/";

 62 /* '.json' upload route */
 63 app.post('/upload', function(req, res){
 64   var folderChar = addFolderPath();
 65   /* create an incoming form object */
 66   var form = new formidable.IncomingForm();
 67 
 68   /* Specify that we want to allow the user to upload multiple files in a single request */
 69   form.multiples = true;
 70 
 71   /* Store all uploads in the /uploads directory */
 72   form.uploadDir = path.join(__dirname, '/uploads');
 73 
 74   /* Every time a file has been uploaded successfully, rename it to it's orignal name */
 75   form.on('file', function(field, file) {
 76     fs.rename(file.path, path.join(form.uploadDir, file.name), function(){return;});
 77     /* Creates directory for uploaded graph */
 78     nodeCmd.get('mkdir -p uploads' + folderChar + file.name.split(".")[0] + folderChar, function(data, err, stderr) {
 79       if (!err)
 80       {
 81         // console.log("data from python script " + data);
 82         /* Assign variable with file name for later coarsening */
 83         fileName = file.name;
 84

 86         if(file.name.split(".")[1] === "gml")
 87         {
 88           nodeCmd.get('python mob' + folderChar + 'gmlToJson2.py uploads' + folderChar + file.name + ' uploads' + folderChar + file.name.split(".")[0] + folderChar + file.name.split(".")[0] + '.json', function(data, err, stderr) {
 89                             if (!err)
 90                             {
 91                               // console.log("data from python script " + data);
 92                               fs.readFile(form.uploadDir + folderChar + file.name.split(".")[0] + folderChar + file.name.split(".")[0] + ".json", 'utf8', function(err, data){
 93                                 if(err)
 94                                 {
 95                                   return console.log(err);
 96                                 }
 97                                 else
 98                                 {
 99                                   /* Store graph size */
100                                   graphSize = JSON.parse(data).graphInfo[0].vlayer;
101                                   /* Send data to client */
102                                   res.end(addNumberOfEdgesToJSON(data));
103                                 }
104                               });
105                             }

111         }
112         else if(file.name.split(".")[1] === "json")
